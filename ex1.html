in2post.y 
    %{
    #include <stdio.h>
    %}
    %token DIGIT PLUS STAR
    %%
    start : expr '\n'           {printf("\nComplete");exit(1);}
	    ;
    expr:  expr PLUS expr        {printf("+ ");}
	    | expr STAR expr     {printf("* ");}
	    | '(' expr ')'
	    | DIGIT             {printf("%d ",$1);}
	    ;
    %%
    
    yyerror()
    {
	    printf("Error");
    }
    main()
    {
	    yyparse();
	    return 1;
    }
